void fill(double array[289][25])
{
	for(int i=0;i<289;i++)
	{
		for(int j=0;j<25;j++)
		{
			array[i][j] = double(rand())/double(RAND_MAX);
		}
	}
}

int get_max(double array[25])
{
	double max=0.0;
	int address=0;
	for(int i=0;i<25;i++)
	{
		if(max < array[i])
		{
			max=array[i];
			address=i;
		}
	}
	array[address]=-8888.88;

	for(i=0;i<25;i++)
	{
		if(i!=address && array[i]>=0)
			array[i]=0.0;
	}
	return address;
}
